数组操作

# 初始化
$ array=([1]="a" [2]="b")

# ❤ 打印 ❤
$ echo $array
a b
# 当做多个元素
$ echo $array[@]
a b
# 当做一个字符串
$ echo $array[*]
a b

# 添加修改元素【注意】下标从 1 开始
$ array[1]="aaa"

# ❤ 数组长度 ❤
$ echo ${#array[@]}
2
$ echo ${#array[*]}
2

# ❤❤ 追加元素 ❤❤ 
$ array[${#array[@]}+1]="c"
$ array[${#array[@]}+1]="d"
$ echo ${array[@]}
aaa b c d
$ echo ${#array[@]}
4
# 批量追加
$ array=("${array[@]}" 111 222 333 444)
$ echo ${array[@]}
aaa b c d 111 222 333 444


# ❤❤❤ 遍历数组 ❤❤❤
for item in ${array[@]}
do
    echo ">>> ${item};"
done
>>> aaa;
>>> b;
>>> c;
>>> d;
>>> 111;
>>> 222;
>>> 333;
>>> 444;
